python paxos/send.py 1 2 paxos/config/local1 -n 1 -c 0

# Start master with -fc flag to drop first occurence of x from master to client
python startMaster.py 3 1 config/singleNodeLocal -fc x

# Start two clients
python startClient config/clientLocal
python startClient config/clientLocal
